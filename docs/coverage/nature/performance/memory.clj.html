001  (ns nature.performance.memory
002    (:require [nature.initialization-operators :as io]
003              [clj-memory-meter.core :as mm]
004              [clojure.test :refer :all]))
005  
006  (deftest generate-sequence-footprint
007    (testing "Ensure memory usage grows sustainably, and report usage to user"
008      (let [length-b-10  (mm/measure (doall (io/generate-sequence [1 0] 10))) ; doall  to force sequences to actually be built
009            length-b-50  (mm/measure (doall (io/generate-sequence [1 0] 50)))
010            length-b-100 (mm/measure (doall (io/generate-sequence [1 0] 100)))
011            length-b-500 (mm/measure (doall (io/generate-sequence [1 0] 500)))
012            length-s-10  (mm/measure (doall (io/generate-sequence ["1" "0"] 10)))
013            length-s-50  (mm/measure (doall (io/generate-sequence ["1" "0"] 50)))
014            length-s-100 (mm/measure (doall (io/generate-sequence ["1" "0"] 100)))
015            length-s-500 (mm/measure (doall (io/generate-sequence ["1" "0"] 500)))]
016        (is (nil? (println "io/generate-sequence memory footprint (Binary), Length 10: " length-b-10)))
017        (is (nil? (println "io/generate-sequence memory footprint (Binary), Length 50: " length-b-50)))
018        (is (nil? (println "io/generate-sequence memory footprint (Binary), Length 100: " length-b-100)))
019        (is (nil? (println "io/generate-sequence memory footprint (Binary), Length 500: " length-b-500)))
020        (is (nil? (println "-----------------------")))
021        (is (nil? (println "io/generate-sequence memory footprint (Singleton String), Length 10: " length-b-10)))
022        (is (nil? (println "io/generate-sequence memory footprint (Singleton String), Length 50: " length-b-50)))
023        (is (nil? (println "io/generate-sequence memory footprint (Singleton String), Length 100: " length-b-100)))
024        (is (nil? (println "io/generate-sequence memory footprint (Singleton String), Length 500: " length-b-500)))
025        (is (nil? (println "-----------------------"))))))
