001  (ns nature.population-operators
002    "Functions that span or operate against entire populations"
003    (:require [nature.population-presets :as pp]
004              [bigml.sampling.simple :as bss])
005    (:gen-class))
006  
007  (defn keep-elite
008    "Find the top `number-to-keep` individuals in `population`, and increment their ages by 1"
009    [population number-to-keep]
010    (map #(update % :age inc) (take number-to-keep (sort-by :fitness-score #(> %1 %2) population))))
011  
012  (defn weighted-selection-of-population
013    "Pick `total-retreived` individuals from `population` with a relative probability of
014    the individual's fitness score divided by the population's aggregate fitness score"
015    ([population total-retrieved]
016     (take total-retrieved (bss/sample population :weigh #(:fitness-score %) :replace true)))
017    ([population total-retrieved replace?]
018     (take total-retrieved (bss/sample population :weigh #(:fitness-score %) :replace replace?))))
019  
020  (defn advance-generation
021    "Apply the functions in `binary-operator-set` until a sufficiently large population is built.
022     Then apply functions in `unary-operator-set` to the result.
023     Optionally, include a map of `settings` to guide overall behavior.
024     If the `:carry-over` setting is added, the elite member `n` of the prior generation will be advanced to the next generation."
025    ([population binary-operator-set unary-operator-set]
026     (let [binary-pop (repeatedly (count population) #(apply (rand-nth binary-operator-set) [(weighted-selection-of-population population 2)]))]
027       (map #((rand-nth unary-operator-set) %) binary-pop)))
028    ([population binary-operator-set unary-operator-set settings]
029    ;; Currently, :carry-over is the only optional behavior
030     (if (> (:carry-over settings) 0)
031       (concat (advance-generation population binary-operator-set unary-operator-set) (keep-elite population (:carry-over settings)))
032       (advance-generation population binary-operator-set unary-operator-set))))
