001  (ns nature.core
002    (:require [nature.spec :as s]
003              [nature.population-presets :as pp]
004              [nature.initialization-operators :as io]
005              [nature.genetic-operators :as go]
006              [nature.population-operators :as po])
007    (:gen-class))
008  
009  (defn evolve
010    "Create and evolve a population under the specified conditions until a termination criteria is reached
011    `allele-set` is a collection of legal genome values
012    `genome-length` is the enforced size of each genetic sequence
013    `population-size` is the enforced number of individuals that will be created
014    `generations` is the number of iterations the algorithm will cycle through
015    `fitness-function` is a partial function accepting generated sequences to evaluate solution qualities
016    `binary-operators` is a collection of partial functions accepting and returning 1 or more individuals
017    `unary-operators` is a collection of partial functions accepting and returning exactly 1 individual
018    `options` an optional map of pre-specified keywords to values that further tune the behavior of nature.
019        Current examples follow:
020        `:carry-over` an integer representing the top n individuals to be carried over between each generation. Default is 1
021        `:solutions` an integer representing the top n individuals to return after evolution completes. Default is 1"
022    ([allele-set genome-length population-size generations fitness-function binary-operators unary-operators]
023     (evolve allele-set genome-length population-size generations fitness-function binary-operators unary-operators {:solutions 1, :carry-over 1}))
024    ([allele-set genome-length population-size generations fitness-function binary-operators unary-operators options]
025     (let [solutions (max 1 (:solutions options))
026           carry-over (max 1 (:carry-over options))]
027       (loop [population (io/build-population population-size allele-set genome-length fitness-function)
028              current-generation 0]
029         (if (>= current-generation generations)
030           (take solutions (sort-by :fitness-score #(> %1 %2) population))
031           (recur (po/advance-generation population binary-operators unary-operators {:carry-over carry-over}) (inc current-generation)))))))
032  
033  (defn -main
034    "I don't do a whole lot ... yet."
035    [& args]
036    (println (evolve pp/binary-genome
037                     pp/default-sequence-length
038                     pp/default-population-size
039                     pp/default-generation-count
040                     pp/sum-alleles
041                     [(go/crossover pp/sum-alleles)]
042                     [(partial go/mutation-operator pp/sum-alleles pp/binary-genome 1)]
043                     {:solutions 10, :carry-over 5})))
