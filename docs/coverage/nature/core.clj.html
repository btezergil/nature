001  (ns nature.core
002    (:require [nature.spec :as s]
003              [nature.population-presets :as pp]
004              [bigml.sampling.simple :as bss])
005    (:gen-class))
006  
007  (defn uuid
008    "More idiomatic wrapper around Java's v1 UUID functionality"
009    []
010    (str (java.util.UUID/randomUUID)))
011  
012  (defn generate-sequence
013    "Creates a genetic sequence of `sequence-length` elements,
014    where each item is in the collection of `alleles`"
015    [alleles sequence-length]
016    (repeatedly sequence-length #(rand-nth alleles)))
017  
018  (defn build-individual
019    "Create a generated individual"
020    ([genetic-sequence fitness-function]
021     (assoc {}
022            :genetic-sequence genetic-sequence
023            :guid (uuid)
024            :parents pp/initializer-name
025            :age pp/default-age
026            :fitness-score (fitness-function genetic-sequence)))
027  
028    ([genetic-sequence parent-coll age fitness-function]
029     (assoc {}
030            :genetic-sequence genetic-sequence
031            :guid (uuid)
032            :parents parent-coll
033            :age age
034            :fitness-score (fitness-function genetic-sequence))))
035  
036  (defn build-population
037    [population-size alleles sequence-length fitness-function]
038    (repeatedly population-size #(build-individual (generate-sequence alleles sequence-length) fitness-function)))
039  
040  (defn weighted-selection-of-population
041    ([population total-retrieved]
042     (take total-retrieved (bss/sample population :weigh #(:fitness-score %) :replace true)))
043    ([population total-retrieved replace?]
044     (take total-retrieved (bss/sample population :weigh #(:fitness-score %) :replace replace?))))
045  
046  (defn -main
047    "I don't do a whole lot ... yet."
048    [& args]
049    (let [population (build-population 20 [0 1] 10 (partial apply +))]
050      (println (loop [new-population '()]
051                 (if (>= (count new-population) (count population))
052                   new-population
053                   (recur (concat new-population (weighted-selection-of-population population 4))))))))
