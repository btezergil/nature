001  (ns nature.genetic-operators
002    (:require [nature.core :as nature]
003              [nature.population-presets :as pp])
004    (:gen-class))
005  
006  (defn fitness-based-scanning-allele
007    "Pick one of two alleles"
008    [allele-1 allele-2 percent-1 percent-2]
009    (if (< percent-1 (rand-int 100))
010      allele-1
011      allele-2))
012  
013  (defn fitness-based-scanning-genome
014    [individual-1 individual-2 percent-1 percent-2]
015    (loop [new-genome '()
016           ind-1      individual-1
017           ind-2      individual-2]
018      (if (empty? ind-1)
019        new-genome
020        (recur (cons (fitness-based-scanning-allele (first ind-1) (first ind-2) percent-1 percent-2) new-genome)
021               (rest ind-1)
022               (rest ind-2)))))
023  
024  (defn fitness-based-scanning
025    [individual-1 individual-2 fitness-function]
026    (let [fitness-total (+ (:fitness-score individual-1) (:fitness-score individual-2))
027          percent-1 (* 100 (/ (:fitness-score individual-1) fitness-total))
028          percent-2 (- 100 percent-1)]
029      (nature/build-individual
030       (fitness-based-scanning-genome (:genetic-sequence individual-1)
031                                      (:genetic-sequence individual-2)
032                                      percent-1
033                                      percent-2)
034       (vector (:guid individual-1) (:guid individual-2))
035       pp/default-age
036       fitness-function)))
037  
038  (defn crossover
039    [individual-1 individual-2 fitness-function]
040    (let [crossover-point (rand-int (count individual-1))
041          split-1 (split-at crossover-point (:genetic-sequence individual-1))
042          split-2 (split-at crossover-point (:genetic-sequence individual-2))
043          parents (vector (:guid individual-1) (:guid individual-2))]
044      (list (nature/build-individual (concat (first split-1) (last split-2))
045                                     parents
046                                     pp/default-age
047                                     fitness-function)
048            (nature/build-individual (concat (first split-2) (last split-1))
049                                     parents
050                                     pp/default-age
051                                     fitness-function))))
